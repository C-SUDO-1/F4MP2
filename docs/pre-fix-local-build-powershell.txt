Pre-fix local PowerShell build sequence

Set-Location "<REPO_ROOT>"
if (Test-Path .\build) { Remove-Item .\build -Recurse -Force }
New-Item -ItemType Directory -Path .\build | Out-Null

cmake -S . -B .\build -G "Visual Studio 17 2022" -A x64 2>&1 | Tee-Object .\cmake-local.log
cmake --build .\build --config RelWithDebInfo --target fo4_coop_service 2>&1 | Tee-Object .\build-service-local.log
cmake --build .\build --config RelWithDebInfo --target plugin_stub 2>&1 | Tee-Object .\build-plugin-local.log

Get-ChildItem .\build -Recurse -Filter *.vcxproj | Select-Object -ExpandProperty BaseName
Get-ChildItem .\build -Recurse -Filter *.dll | Sort-Object LastWriteTime -Descending | Select-Object LastWriteTime, Length, FullName
